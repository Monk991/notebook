- app/app.vue

```vue
<script setup lang="ts">
import { Button } from "@/components/ui/button";

const colorMode = useColorMode();
function toggleDark() {
  colorMode.preference = colorMode.value === "dark" ? "light" : "dark";
}
</script>

<template>
  <Button variant="ghost"  @click="toggleDark">
     Toggle theme
  </Button>
</template>
```
